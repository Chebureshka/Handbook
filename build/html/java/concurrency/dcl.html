class Singleton {

    private volatile static Singleton INSTANCE;

    private Singleton() {}

    public static Singleton getInstance() {
        Singleton localRef = INSTANCE;

        if (localRef == null) {
            synchronized (Singleton.class) {
                localRef = INSTANCE;

                if (localRef == null) {
                    localRef = new Singleton();
                    INSTANCE = localRef;
                }
            }
        }

        return localRef;
    }
}
